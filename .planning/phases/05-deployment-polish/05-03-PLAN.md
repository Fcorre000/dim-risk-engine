```python
import pandas as pd, numpy as np, io

def make_sample_invoice(n=5000) -> bytes:
    rng = np.random.default_rng(42)
    df = pd.DataFrame({
        "Master Tracking Number": [f"TRK{i:07d}" for i in range(n)],
        "Service Type": rng.choice(["FedEx Ground", "FedEx Express", "FedEx Home Delivery"], n),
        "Pricing Zone": rng.choice(["02", "04", "06", "08"], n),
        "Billed Weight": rng.uniform(1, 150, n).round(1),
        "Net Charge": rng.uniform(5, 500, n).round(2),
        "DIM Flag": rng.choice(["Y", "N"], n, p=[0.3, 0.7]),
        "Transaction Type": ["NonDocument"] * n,
        # Add remaining columns as zeros — reindex(fill_value=0) in ingest handles this
    })
    buf = io.BytesIO()
    df.to_csv(buf, index=False)
    return buf.getvalue()
```
